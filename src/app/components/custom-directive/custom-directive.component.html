  export class ProductFormGroup extends FormGroup {{ '{' }}
    constructor() {{ '{' }}
      super({{ '{' }}
        name: new ProductFormControl('Name','name', "", Validators.required),

            Validators.minLength(3),
            Validators.maxLength(10)])),
        price: new ProductFormControl('Price', 'price', '',
